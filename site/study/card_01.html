네번 째 : undefined

변수를 선언하고 데이터 값은 지정하지 않았을 때 결과 값으로 나타나는 데이터

var temp1 = 10;
var temp2;

console.log(temp1); // 결과 값 : 10
console.log(temp2); // 결과 값 : undefined

다섯번 째 : null

변수를 빈 상태로 초기화 하거나, 데이터는 저장했지만 값이 존재하지 않을 때 null값으로 반환 한다.

var obj = 10;
obj = null;
console.log(obj); // 결과 값 : null | 변수를 빈 상태로 만든다.

var obj = document.getElementById('gnb');
console.log(obj); // 결과 값 : 변수에 데이터 값을 저장했으나, 값이 존재하지않아 null을 저장
