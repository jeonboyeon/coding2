01. if 문

{
    //true, 1 이라 쓰면 true가 나오고, false, 0, undefined, null 을 쓰면 false가 나온다.
    //문자열 "0 ~", [0](배열), {a: 100}(객체) 를 쓰면 true가 나온다.
    if("조건식"){ //true(참 출력)
        document.write("실행되었습니다(true).");
    } else { //false(거짓 출력)
        document.write("실행되었습니다(false).");
    }
}
